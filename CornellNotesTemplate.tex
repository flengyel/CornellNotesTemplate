\documentclass{article} 
\usepackage{calc} 
\usepackage[top=0.75in,bottom=0.25in,left=0.5in,right=0.5in]{geometry} 
\usepackage{tikz} 
\usetikzlibrary{calc}

\newlength{\wholeboxwd}
    \setlength{\wholeboxwd}{.99\textwidth}
\newlength{\wholeboxht}
    \setlength{\wholeboxht}{0.95\textheight}
% step = cgridht / 27
\newlength{\cuewd}    \setlength{\cuewd}{0.309\wholeboxwd}
\newlength{\summht}   \setlength{\summht}{0.206\wholeboxht}
\newlength{\cgridht}  \setlength{\cgridht}{\wholeboxht-\summht}
\newlength{\cgridwd}  \setlength{\cgridwd}{\wholeboxwd-\cuewd}
\newlength{\mystep} \setlength{\mystep}{ \cgridht / 27}
\newlength{\xorig}
\newlength{\yorig}
\setlength{\xorig}{0cm}
\setlength{\yorig}{0cm}

\begin{document} 
\thispagestyle{empty}
\begin{center}
\begin{tikzpicture} 

\foreach \i in {1,2,...,27}{
\draw[gray!75,thin] ($(\cuewd,\wholeboxht)+(0,-\mystep*\i)$) -- ($(\wholeboxwd,\wholeboxht)+(0,- \mystep*\i)$);
}

% Draw the horizontal summary section separator
\draw [line width=.2pt] (\xorig,\summht) -- (\wholeboxwd,\summht);  
% Draw vertical the cue section separator
\draw [line width=.2pt] (\cuewd,\summht) -- (\cuewd,\wholeboxht);
% Draw the bounding box
\draw (\xorig,\yorig) rectangle (\wholeboxwd,\wholeboxht);

\end{tikzpicture} 
\end{center}

\end{document}
